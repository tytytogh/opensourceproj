struct person {
	//신체 정보

	//성별
	char gender;
	//나이
	int age;
	//체중과 신장 저장할 변수
	double weight;
	double height;
};

double tdee(person p, int activeMeta_in) {
	//하루 총 에너지 소비량 추정치 계산
	//2020 한국인 영양소 섭취기준의 EER 산출공식에 따름

	//활동대사량 비율 저장할 배열
	double active[4][4] = { { 1.0,1.13,1.26,1.42 }, 
							{ 1.0,1.11,1.25,1.56 }, 
							{ 1.0,1.16,1.31,1.48 }, 
							{ 1.0,1.12,1.27,1.45 } };
	//계산 결과 저장 변수
	double result;

	if (p.gender == 'm' || p.gender == 'M') { //남성일 때
		if (p.age < 20 && p.age > 2) {
			//나이가 3살부터 19살 사이
			result = 88.5 - (61.9 * p.age) + active[0][activeMeta_in - 1] * (26.7 * p.weight + 903 * p.height / 100);
		}
		else if (p.age >= 20) {
			//20세 이상
			result = 662 - (9.53 * p.age) + active[1][activeMeta_in - 1] * (15.91 * p.weight + 539.6 * p.height / 100);
		}
		else {
			//예외
			return 0;
		}
	}
	else if (p.gender == 'w' || p.gender == 'W') { //여성일 때
		if (p.age < 20 && p.age > 2) {
			//3~19세
			result = 135.3 - (30.8 * p.age) + active[2][activeMeta_in - 1] * (10.0 * p.weight + 934 * p.height / 100);
		}
		else if (p.age >= 20) {
			//20세 이상
			result = 354 - (6.91 * p.age) + active[3][activeMeta_in - 1] * (9.36 * p.weight + 726 * p.height / 100);
		}
		else {
			//예외
			return 0;
		}
	}
	else {
		//예외
		return 0;
	}

	return result;
}

void filemanage(int days, double Calcalculator) {
	//양수 days가 들어옴

	//식단에 음식을 추가할 방법
	int way;
	//음식이름 입력 받을 string 객체
	string inhand_food;
	//음식 칼로리 입력 받을 변수
	int inhand_cal;
	//***********종류별로 각각 따로 텍스트를 분리할 string 객체 필요할 듯
	//저장된 음식 이름 파일입력 스트림 관리 객체
	ifstream infood;
	//저장된 음식별 그램당 칼로리 파일입력 스트림 관리 객체
	ifstream incal;
	//foodtable에서 가져올 직접 입력한 음식 저장할 string 객체
	string foodinhand;
	//직접 입력한 음식이름과 칼로리를 각 텍스트 파일에 저장하기 위한 객체
	ofstream inhand;
	//foodtable에서 가져올 텍스트 저장 공간
	string food;
	//calories에서 가져올 텍스트 저장 공간
	string cal;

	//파일 오픈(ifstream 디폴트인 iso::in 모드)
	infood.open("foodtable.txt");
	incal.open("calories.txt");
	//파일이 올바르지 않다면 종료
	if (infood.fail() || incal.fail()) {
		cout << "파일 오류입니다" << endl;
		cout << "메뉴로 돌아갑니다" << endl;
		system("pause");
		return;
	}

	//파일 포인터를 끝으로 옮김
	infood.seekg(0, ios::end);
	incal.seekg(0, ios::end);
	//파일 크기 저장(파일의 각 줄의 끝에는 \r과 \n이 들어가있음)
	int sizefood = infood.tellg();
	int sizecal = incal.tellg();
	//다시 파일 포인터 원상복구
	infood.seekg(0, ios::beg);
	incal.seekg(0, ios::beg);
	//'\r' 개수 셀 변수
	int countfood = 0;
	int countcal = 0;

	//각 파일의 텍스트를 객체로 옮김(스페이스로 줄 구분x)
	getline(infood, food);
	getline(incal, cal);

	//음식이름, 칼로리 텍스트 중 \r 개수 세기
	for (int i = 0; i < sizefood; i++) { //******************위에서 food 스트링 객체에 5개의 원소가 있으므로, 그 이상 읽으려고해서 오류 남
		if (food[i]=='\r') {
			countfood++;
		}
	}
	for (int i = 0; i < sizecal; i++) {
		if (cal[i] == '\r') {
			countcal++;
		}
	}
	//음식 이름과 칼로리 텍스트 길이
	int foodlength = sizefood - (2 * countfood);
	int callength = sizecal - (2 * countcal) + 1;
	//각 종류를 저장할 string 객체를 저장할 vector(size는 텍스트 size)
	vector<string> f(foodlength);
	vector<string> c(callength);
	//각 벡터의 string에 텍스트 파일에서 읽어오기
	for (int i = 0; i < foodlength; i++) {
		getline(infood, f[i]);
	}
	for (int i = 0; i < callength; i++) {
		getline(incal, c[i]);
	}
	for (int i = 0; i < foodlength; i++) {
		cout << f[i] << endl;
	}
	//파일에서 음식과 음식 칼로리를 받아오기
	


	for (int i = 0; i < days; i++) {
		cout << "식단짜는 방법 선택(1과 2중 선택) ▷ ";
		cin >> way;
		if (way == 1) {
			//종류에서 선택

		}
		else if (way == 2) {
			//직접 입력
			cout << "음식 이름을 입력하세요(영어로) ▷ ";
			//음식 이름 저장
			cin >> inhand_food;
			inhand << inhand_food;
			cout << "음식의 드신 양의 칼로리를 입력해주세요";
			//*****입력받은 칼로리를 calories 파일에 저장해야함 -> 반드시 그램당 칼로리일 것!
			//incal << inhand_cal;
		}
		else {
			//예외
			i--;
			continue;
		}
	}

	//파일 닫기
	infood.close();
	incal.close();

	return;
}
